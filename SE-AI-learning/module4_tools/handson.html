Step FC-1

Step FC-2

tool_calls を解析して実際の関数を実行する

LLM が返した tool_calls の関数名と引数を解析し、対応するPython関数を実行してください。

# Step FC-2: tool_calls の解析と関数実行
import json

# モックの検索関数（実際の演習ではDBアクセスに置き換え）
def search_ticket(keyword: str, status: str = "open") -> list[dict]:
    mock_db = [
        {"id": "T-001", "title": "ログインエラー発生", "status": "open"},
        {"id": "T-002", "title": "パスワードリセット不可", "status": "open"},
        {"id": "T-003", "title": "ログイン後に画面がフリーズ", "status": "closed"},
    ]
    return [t for t in mock_db if keyword in t["title"] and t["status"] == status]

# tool_calls レスポンスの処理
msg = response.choices[0].message
if msg.tool_calls:
    tc = msg.tool_calls[0]
    args = json.loads(tc.function.arguments)  # EDIT HERE!
    result = search_ticket(**args)              # EDIT HERE!

    print(f"検索結果: {result}")

    # 結果を role: "tool" として LLM に返す
    messages = [  # EDIT HERE!
        {"role": "user", "content": "ログインエラーのチケットを探して"},
        msg,  # LLM の tool_calls メッセージ
        {
            "role": "tool",
            "tool_call_id": tc.id,
            "content": json.dumps(result, ensure_ascii=False)
        }
    ]

    final_response = client.chat.completions.create(
        model=MODEL,
        messages=messages
    )
    print(final_response.choices[0].message.content)
ヒント ①
tc.function.arguments は JSON 文字列です
json.loads(tc.function.arguments) で辞書に変換
search_ticket(**args) でアンパックして関数に渡します
ヒント ②
{"role": "tool", "tool_call_id": tc.id, "content": json.dumps(result)}
tool_call_id は tc.id から取得します（どのtool_callへの応答かを紐付けるため）
messages リスト: [ユーザーメッセージ, LLMのtool_callsメッセージ, ツール結果メッセージ]

Step FC-3

複数ツールを登録してエージェントループを実装する

チケット検索と経費上限照会の2つのツールを登録し、LLMが必要に応じて自律的にツールを選んで使うエージェントループを実装してください。

# Step FC-3: 複数ツールとエージェントループ
tools = [search_ticket_tool, get_expense_limit_tool]  # 2つのツールを登録

def run_agent(user_message: str) -> str:
    messages = [{"role": "user", "content": user_message}]

    while True:
        response = client.chat.completions.create(
            model=MODEL,
            tools=tools,
            messages=messages
        )
        msg = response.choices[0].message

        # ツール呼び出しがなければ終了
        if not msg.tool_calls:  # EDIT HERE!
            return msg.content

        # 全ての tool_calls を処理する
        messages.append(msg)
        for tc in msg.tool_calls:  # EDIT HERE!
            args = json.loads(tc.function.arguments)
            if tc.function.name == "search_ticket":
                result = search_ticket(**args)
            elif tc.function.name == "get_expense_limit":
                result = get_expense_limit(**args)
            else:
                result = {"error": "Unknown tool"}

            messages.append({
                "role": "tool",
                "tool_call_id": tc.id,
                "content": json.dumps(result, ensure_ascii=False)
            })

# テスト実行
print(run_agent("交通費の上限とオープン中のログイン関連チケット数を教えて"))
ヒント
while ループ: tool_calls がなくなるまでLLMを繰り返し呼び出します
msg.tool_calls が空（None または []）なら最終回答として返します
複数の tool_calls がある場合は for tc in msg.tool_calls: で全て処理します
# 解答: Step FC-3（エージェントループの核心部分）

    response = client.chat.completions.create(model=MODEL, tools=tools, messages=messages)
    msg = response.choices[0].message

    for tc in msg.tool_calls:

Step MCP-1

FastMCP でMCPサーバーを実装する

FastMCP を使って社内ツール（チケット検索・経費上限照会）をMCPサーバーとして公開してください。

# Step MCP-1: FastMCP サーバーの実装
from fastmcp import FastMCP

mcp = FastMCP("Axxx Japan-社内ツールサーバー")

@mcp.tool()  # EDIT HERE!
def search_ticket(keyword: str, status: str = "open") -> dict:
    """チケットシステムで検索します。keyword: 検索キーワード、status: open/closed"""
    mock_db = [
        {"id": "T-001", "title": "ログインエラー", "status": "open"},
        {"id": "T-002", "title": "パスワードリセット不可", "status": "open"},
    ]
    results = [t for t in mock_db if keyword in t["title"] and t["status"] == status]
    return {"tickets": results, "count": len(results)}

@mcp.tool()  # EDIT HERE!
def get_expense_limit(category: str) -> dict:
    """経費カテゴリの上限金額を返します。category: 交通費/宿泊費/会議費"""
    limits = {"交通費": 50000, "宿泊費": 15000, "会議費": 30000}
    limit = limits.get(category)
    if limit is None:
        return {"error": f"カテゴリ '{category}' は存在しません"}
    return {"category": category, "limit": limit, "currency": "JPY"}

if __name__ == "__main__":
    mcp.run(transport="streamable-http", host="127.0.0.1", port=8000)
ヒント
通常の Python 関数に @mcp.tool() を付けるだけでツールとして公開されます
関数の型アノテーション（str, int など）が自動的に JSON Schema に変換されます
docstring がツールの description として使われます
