Step 1

ルールベース評価 ─ JSON形式の出力が仕様通りかチェックする

evaluate_json_output() 関数を完成させてください。 LLMが返したJSON文字列が EXPECTED_SCHEMA の仕様を満たすかを検証します。 以下の4つのチェックを実装してください。

  1. llm_output が有効な JSON かどうか（json.loads でパース）
  2. EXPECTED_SCHEMA の required_keys がすべて存在するか
  3. priority が priority_values のいずれかか
  4. action_items が最低 action_items_min_count 件あるか
EXPECTED_SCHEMA = {
    "required_keys": ["summary", "priority", "action_items"],
    "priority_values": ["high", "medium", "low"],
    "action_items_min_count": 1,
}

def evaluate_json_output(llm_output: str) -> dict:
    result = {"passed": False, "errors": [], "warnings": []}
        # EDIT HERE! 4つのチェックを実装してください
    return result
ヒント
まず json.loads(llm_output) を try/except json.JSONDecodeError で囲みます。失敗したら errors に追加して return result します
required_keys のチェックは for key in EXPECTED_SCHEMA["required_keys"]: でループし、key not in data のときにエラーを追加します
priority_values チェックは data.get("priority") not in EXPECTED_SCHEMA["priority_values"] で判定します
すべてのチェックを通過したら result["passed"] = True にします
def evaluate_json_output(llm_output: str) -> dict:
    result = {"passed": False, "errors": [], "warnings": []}

    # チェック1: 有効なJSONかどうか
    try:
        data = json.loads(llm_output)
    except json.JSONDecodeError as e:
        result["errors"].append(f"JSONパースエラー: {e}")
        return result

    # チェック2: 必須キーの存在確認
    for key in EXPECTED_SCHEMA["required_keys"]:
        if key not in data:
            result["errors"].append(f"必須キーが存在しない: {key}")

    # チェック3: priority の値が有効か確認
    if "priority" in data:
        if data["priority"] not in EXPECTED_SCHEMA["priority_values"]:
            result["errors"].append(
                f"priority の値が無効: {data['priority']} "
                f"(有効値: {EXPECTED_SCHEMA['priority_values']})"
            )

    # チェック4: action_items の件数確認
    if "action_items" in data:
        min_count = EXPECTED_SCHEMA["action_items_min_count"]
        if len(data["action_items"]) < min_count:
            result["errors"].append(f"action_items が {min_count} 件未満")

    result["passed"] = len(result["errors"]) == 0
    return result

Step 2

LLM-as-judge ─ 回答品質を別のLLMで5段階採点する

llm_judge() 関数を完成させてください。 JUDGE_PROMPT_TEMPLATE に質問と回答を埋め込み、 LLM に採点させて JSON 形式の評価結果を返します。 

JUDGE_PROMPT_TEMPLATE = """
# タスク
以下のAI回答の品質を評価してください。

# 評価基準
- 正確性: 情報が正確か（1-5）
- 簡潔性: 不要な情報がないか（1-5）
- 実用性: ユーザーの問題を解決できるか（1-5）

# 質問
{question}

# AI回答
{answer}

# 出力形式（必ずこのJSON形式で返してください）
{{
  "scores": {{"accuracy": <1-5>, "conciseness": <1-5>, "usefulness": <1-5>}},
  "overall": <1-5>,
  "reason": "<評価の根拠を1-2文で>"
}}
"""

        # 返ってきた JSON をパースして返してください
    return {}
ヒント
response = client.responses.create(model=model, input=judge_prompt) でAPIを呼び出します
response.output_text に文字列が返ってきます。json.loads() でパースします
LLMが ```json ... ``` のようなコードブロックで返す場合があります。re.search(r'\{.*\}', text, re.DOTALL) でJSONを抽出する方法が有効です
パースエラーに備えて try/except で囲み、エラーの場合は空辞書を返しましょう
def llm_judge(client, model: str, question: str, answer: str) -> dict:
    judge_prompt = JUDGE_PROMPT_TEMPLATE.format(question=question, answer=answer)
    try:
        response = client.responses.create(model=model, input=judge_prompt)
        text = response.output_text
        # コードブロックで囲まれている場合に対応する
        match = re.search(r'\{.*\}', text, re.DOTALL)
        if match:
            return json.loads(match.group())
        return json.loads(text)
    except Exception as e:
        print(f"llm_judge エラー: {e}")
        return {}

Step 3

回帰テスト ─ 評価セットを作成して繰り返し自動テストを行う

run_regression_test() 関数を完成させてください。 GOLDEN_DATASET の各テストケースに対してLLMに回答させ、 expected_keywords が回答に含まれているかで PASS/FAIL を判定します。 

GOLDEN_DATASET = [
    {"id": "T001", "question": "有給休暇は年間何日ですか？",
     "expected_keywords": ["20日", "有給"], "min_score": 3},
    {"id": "T002", "question": "テレワークの申請はどうすればいいですか？",
     "expected_keywords": ["申請", "システム"], "min_score": 3},
    {"id": "T003", "question": "経費精算の締め日はいつですか？",
     "expected_keywords": ["締め日", "月末", "精算"], "min_score": 3},
]

    results = {"total": len(GOLDEN_DATASET), "passed": 0, "failed": 0, "details": []}
    for case in GOLDEN_DATASET:
                # EDIT HERE! 各テストケースについて:
                # 1. client.responses.create() で質問に回答させる
                # 2. expected_keywords が回答に含まれるかチェック
                # 3. 含まれていれば passed++, なければ failed++
                # 4. results["details"] に結果を追加
        pass

ヒント
response = client.responses.create(model=model, input=case["question"]) で各質問に回答させます
answer = response.output_text で回答テキストを取得します
passed = all(kw in answer for kw in case["expected_keywords"]) でキーワード包含チェックができます
results["details"].append({"id": case["id"], "question": case["question"], "passed": passed, "expected_keywords": case["expected_keywords"]}) で詳細を記録します
def run_regression_test(client, model: str) -> dict:
    results = {"total": len(GOLDEN_DATASET), "passed": 0, "failed": 0, "details": []}
    for case in GOLDEN_DATASET:
        try:
            # LLMに質問へ回答させる
            response = client.responses.create(model=model, input=case["question"])
            answer = response.output_text

            # 期待キーワードが回答に含まれるか確認する
            passed = all(kw in answer for kw in case["expected_keywords"])
            results["passed" if passed else "failed"] += 1

            results["details"].append({
                "id": case["id"],
                "question": case["question"],
                "passed": passed,
                "expected_keywords": case["expected_keywords"],
                "answer_preview": answer[:80],
            })
        except Exception as e:
            results["failed"] += 1
            results["details"].append({
                "id": case["id"], "question": case["question"],
                "passed": False, "error": str(e)
            })
    return results
